function deepCount(a){
  let count = a.length;
  for (let i = 0; i < a.length; i++) {
    if (Array.isArray(a[i])) {
      count += deepCount(a[i]);
    }
  }
  return count;
}

              
            
              
function myLanguages(results) {
    let sortable = [];
    for (let score in results) {
        sortable.push([score, results[score]]);
    }
    sortable.sort((a, b) => b[1] - a[1]);
    let resultArr = [];
    sortable.map(item => (item[1] >= 60) ? resultArr.push(item[0]) : null);
    return resultArr;
}
              
            
              
function sortByBit(arr) {
    function compareNumeric(a, b) {
        let bit1 = (a.toString(2).match(/1/g) || []).length;
        let bit2 = (b.toString(2).match(/1/g) || []).length;
        if (bit1 > bit2) return a - b;
    }               
    return arr.sort(compareNumeric);
}
